    scores[i] = get_double("Score: ");
    }

    printf("average: %f\n", average(TOTAL, scores));
}

float average(int length, int array[])
{
    float sum = 0.0;
    for (int i = 0; i < length; i++)
    {
        sum += array[i];
    }
    return sum / (float) length;
}
